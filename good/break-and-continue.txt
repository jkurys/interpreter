int main() {
    int x = 0;
    while(true) {
        if (x == 10) {
            break;
            print("NEVER PRINTED");
        }
        x++;
        print(x);
        print("\n");
    }
    while(x < 20) {
        x++;
        print(x);
        print("\n");
        continue;
        print("NEVER PRINTED");
    }
}
